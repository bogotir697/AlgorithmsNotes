#### [[Вопросы к Экзамену]]
---
### Описание
Список на динамическом массиве это довольно простая структура данных. Она хранит три поля: *размер списка*, *размер массива* и *сам массив*. Важно понять, что размер массива $\not=$ размеру списка. Массив используется для хранения элементов, но они при этом могут занимать не все ячейки.
Поговорим про методы. Существует 4 основных метода: получение размера массива и получение, добавление и удаление элемента. Получение элемента и размера массива это методы в одно действие, их разбирать нет особого смысла, а вот о добавлении и удалении элемента можно поговорить. 
Добавление элемента разделяется на 2 шага:
1. Проверить, есть ли ещё место в массиве, если нет, создать новый на несколько элементов больше и скопировать старый массив в новый.
2. Переместить все элементы, чьи индексы больше или равны индексу вставляемого элемента, на одну позицию влево.
При удалении элемента всё просто: все элементы массива с индексами больше индекса удаляемого элемента смещаются на одну позицию вправо.
### Реализация
```cpp
#include <iostream>
#include <stdexcept>
using namespace std;
template<typename T>
class Array {
	private:
		int _size = 0;
		int _capacity = 0;
		T* _arr = nullptr;
	public:
		Array() : _size = 0, _capacity = 5 {
			_arr = new T[_capacity];
		}
		~Array() {
			if (_arr) delete _arr[];
			_size = 0;
			_capacity = 0;
		}
		int size() {
			return _size;
		}
		void add(T value, unsigned int i) {
			if (!(i <= _size)) throw runtime_error("Non-valid index error");
			if (!(_size + 1 < _capacity)) {
				_capacity *= 1.5;
				T* newarr = new T[_capacity];
				for (int i = 0; i < _size; i++) {
					newarr[i] = _arr[i];
				}
				newarr[_size] = value;
				delete _arr;
				_arr = newarr;
			}
			for (int j = _size; j > i; j--) {
					_arr[j] = _arr[j - 1];
				}
				_arr[i] = value;
			size++;
		}
		T remove(unsigned int i) {
			if (!(i < _size)) throw runtime_error("Non-valid index error");
			for (int j = i; j < _size - 1; j++) {
				_arr[j] = _arr[j + 1];
			}
			_size--;
		}
		T& get(unsigned int i) {
			if (!(i < _size)) throw runtime_error("Non-valid index error");
			return _arr[i];
		}
};
```
